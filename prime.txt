Programa -> Qhapaq Definicion .

Definicion -> LParen RParen DefinicionPrime .
DefinicionPrime -> LBrace DefinicionPrimeBody .
DefinicionPrimeBody -> Bloque RBrace | RBrace .

Bloque -> ListaCondiciones | Declaraciones | SiqiySentencia | Operaciones .

ListaCondiciones -> Condiciones ListaCondicionesPrime .
ListaCondicionesPrime -> CondicionesPrime ListaCondicionesPrime | Vacio .

Condiciones -> Ari LParen OperacionRelacional RParen LBrace Bloque RBrace CondicionesPrime | CicloFor | CicloWhile .
CondicionesPrime -> CondicionElseif CondicionElse | Vacio .

CicloFor -> For LParen Declaracion PuntoyComa OperacionRelacional PuntoyComa Operacion RParen LBrace Bloque RBrace .

CicloWhile -> While LParen OperacionRelacional RParen LBrace Bloque RBrace .

CondicionElseif -> ManaChayqaAri LParen OperacionRelacional RParen LBrace Bloque RBrace | Vacio .

CondicionElse -> ManaChayqa LBrace Bloque RBrace | Vacio .

OperacionRelacional -> Expresion OperacionRelacionalPrime .
OperacionRelacionalPrime -> OperadorRelacional Expresion | Vacio.

Declaraciones -> Declaracion DeclaracionesPrime .
DeclaracionesPrime -> Declaraciones | Vacio .

Declaracion -> Tipo DeclaracionPrime .
DeclaracionPrime -> Operacion .

SiqiySentencia -> Siqiy LParen ListaExpresion RParen .

Operaciones -> Operacion OperacionesPrime .
OperacionesPrime -> Operacion OperacionesPrime | Vacio .

Operacion -> ID OperacionPrime .
OperacionPrime -> Assign OperacionAritmetica | Vacio.

OperacionAritmetica -> Expresion OperadorAritmetico Expresion .

Expresion -> Termino ExpresionPrime .
ExpresionPrime -> OperadorRelacional Termino ExpresionPrime | Vacio .

Termino -> Factor TerminoPrime .
TerminoPrime -> OperadorAritmetico Factor TerminoPrime | Vacio .

Factor -> LParen Expresion RParen | ID | NUMERO | CADENA | Yanqa | Chiqaq .

Tipo -> Yupay | Chunkayuq | Sananpa | Qaytu | Bool .

ListaExpresion -> Expresion ListaExpresionPrime .
ListaExpresionPrime -> Mas Expresion ListaExpresionPrime | Vacio .

OperadorAritmetico -> Mas | Menos | Multiplica | Divide .
OperadorRelacional -> Mayor | Menor | Igual | Diferente | MayorIgual | MenorIgual .

Mas -> + .
Menos -> menor .
Multiplica -> * .
Divide -> / .
LParen -> ( .
RParen -> ) .
LBrace -> lbrace .
RBrace -> rbrace .
Assign -> asignacion .
Mayor -> mayor .
Menor -> menor .
Igual -> igual .
Diferente -> diferente .
MenorIgual -> menorigual .
MayorIgual -> mayorigual .
PuntoyComa -> ; .
Ari -> ari .
ManaChayqaAri -> mana_chayqa_ari .
ManaChayqa -> mana_chayqa .
For -> imapaq .
While -> chaykama .
ID -> id .
NUMERO -> numero .
CADENA -> cadena .
Qhapaq -> qhapaq .
Yanqa -> yanqa .
Chiqaq -> chiqaq .
Siqiy -> siqiy .
Vacio -> X .
